cmake_minimum_required(VERSION 3.15)

# Set your project name
project(CommandAliasManager)

# Specify the C++ standard
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED True)

# Add the executable
add_executable(CommandAliasManager main.cpp)  # Replace main.cpp with your source files

# Find the jsoncpp package using vcpkg
find_package(jsoncpp CONFIG REQUIRED)

# Link your executable with jsoncpp
target_link_libraries(CommandAliasManager PRIVATE jsoncpp_lib)
